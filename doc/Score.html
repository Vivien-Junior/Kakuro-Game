def initialize(difficulte, aide)
        case difficulte
                when "facile"
                        if(aide)
                                @nb_score = 15000
                        else
                                @nb_score = 25000
                        end
                when "moyen"
                        if(aide)
                                @nb_score = 25000
                        else
                                @nb_score = 35000
                        end
                when "difficile"
                        if(aide)
                                @nb_score = 40000
                        else
                                @nb_score = 50000
                        end
                else
                        puts "Error"
                end
end
